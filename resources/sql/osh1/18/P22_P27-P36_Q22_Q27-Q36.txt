SELECT
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '1' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '2' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '3' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '4' THEN GRADES.SYS_GUID ELSE NULL END), 
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '5' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '6' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '7' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '8' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '9' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '10' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '11' THEN GRADES.SYS_GUID ELSE NULL END),
	  
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '1' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '2' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '3' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '4' THEN PARTICIPANTS.SYS_GUID ELSE NULL END), 
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '5' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '6' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '7' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '8' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '9' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '10' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASS_PROFILE = '11' THEN PARTICIPANTS.SYS_GUID ELSE NULL END)  
FROM GRADES
INNER JOIN PARTICIPANTS ON PARTICIPANTS.SYS_GUIDFK = GRADES.SYS_GUID
INNER JOIN EGEEXTPARTICIPANTS ON EGEEXTPARTICIPANTS.SYS_GUIDFK = PARTICIPANTS.SYS_GUID
