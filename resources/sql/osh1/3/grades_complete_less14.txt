SELECT 
	   CNT_1 + CNT_2 + CNT_3 + CNT_4 + CNT_5 + CNT_6 + CNT_7 + CNT_8 + CNT_9 + CNT_10 + CNT_11 + CNT_12 + CNT_13 AS "CNT"
FROM (SELECT
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_1_3 < 14 THEN CNT_GRADES_COMPLETE_1_3 ELSE NULL END) AS "CNT_1",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_2_4 < 14 THEN CNT_GRADES_COMPLETE_2_4 ELSE NULL END) AS "CNT_2",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_1_2_3 < 14 THEN CNT_GRADES_COMPLETE_1_2_3 ELSE NULL END) AS "CNT_3",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_1_2_3_4 < 14 THEN CNT_GRADES_COMPLETE_1_2_3_4 ELSE NULL END) AS "CNT_4",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_2_3 < 14 THEN CNT_GRADES_COMPLETE_2_3 ELSE NULL END) AS "CNT_5",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_2_3_4 < 14 THEN CNT_GRADES_COMPLETE_2_3_4 ELSE NULL END) AS "CNT_6",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_1_2 < 14 THEN CNT_GRADES_COMPLETE_1_2 ELSE NULL END) AS "CNT_7",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_1_4 < 14 THEN CNT_GRADES_COMPLETE_1_4 ELSE NULL END) AS "CNT_8",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_3_4 < 14 THEN CNT_GRADES_COMPLETE_3_4 ELSE NULL END) AS "CNT_9",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_1_2_4 < 14 THEN CNT_GRADES_COMPLETE_1_2_4 ELSE NULL END) AS "CNT_10",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_1_3_4 < 14 THEN CNT_GRADES_COMPLETE_1_3_4 ELSE NULL END) AS "CNT_11",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_6_9 < 14 THEN CNT_GRADES_COMPLETE_6_9 ELSE NULL END) AS "CNT_12",
	  COUNT(CASE WHEN CNT_GRADES_COMPLETE_5_6 < 14 THEN CNT_GRADES_COMPLETE_5_6 ELSE NULL END) AS "CNT_13"	  
FROM (SELECT
	  SUM(CASE WHEN COMPLETESET = '1' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_1_3",
	  SUM(CASE WHEN COMPLETESET = '2' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_2_4",
	  SUM(CASE WHEN COMPLETESET = '3' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_1_2_3",
	  SUM(CASE WHEN COMPLETESET = '4' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_1_2_3_4",
	  SUM(CASE WHEN COMPLETESET = '5' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_2_3",
	  SUM(CASE WHEN COMPLETESET = '6' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_2_3_4",
	  SUM(CASE WHEN COMPLETESET = '7' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_1_2",
	  SUM(CASE WHEN COMPLETESET = '8' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_1_4",
	  SUM(CASE WHEN COMPLETESET = '9' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_3_4",
	  SUM(CASE WHEN COMPLETESET = '10' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_1_2_4",
	  SUM(CASE WHEN COMPLETESET = '11' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_1_3_4",
	  SUM(CASE WHEN COMPLETESET = '14' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_6_9",
	  SUM(CASE WHEN COMPLETESET = '16' THEN COUNTER ELSE NULL END) AS "CNT_GRADES_COMPLETE_5_6"	  
FROM (SELECT 
	 		GRADES.GRADEGROUPFK AS "GRADE",
	 		GRADES.LITER AS "LITER",
			GRADES.CLASSCOMPLETESET AS "COMPLETESET",
			COUNT(PARTICIPANTS.SYS_GUID) AS "COUNTER"
	 FROM SCHOOLS
	 INNER JOIN GRADES ON GRADES.SYS_GUIDFK = SCHOOLS.SYS_GUID
	 INNER JOIN PARTICIPANTS ON GRADES.SYS_GUID = PARTICIPANTS.SYS_GUIDFK
	 WHERE GRADES.CLASSCOMPLETE = '1' AND SCHOOLS.TOWNTYPEFK IN ('1','2') AND PARTICIPANTS.PEOPLELEAVING IS NULL
	 GROUP BY GRADES.GRADEGROUPFK,GRADES.LITER,GRADES.CLASSCOMPLETESET) AS T1) AS T2) AS T3
