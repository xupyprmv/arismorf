SELECT
	  GRADES_COMPLETE_1_3 + GRADES_COMPLETE_2_4 + GRADES_COMPLETE_1_2_3 + GRADES_COMPLETE_1_2_3_4 + GRADES_COMPLETE_2_3 + GRADES_COMPLETE_2_3_4 + GRADES_COMPLETE_1_2 + GRADES_COMPLETE_1_4 + GRADES_COMPLETE_3_4 + GRADES_COMPLETE_1_2_4 + GRADES_COMPLETE_1_3_4 AS "GRADE_COMPLETE_14" 	  
FROM (SELECT
	  COUNT(CASE WHEN COMPLETESET = '1' THEN GRADE ELSE NULL END)/2 AS "GRADES_COMPLETE_1_3",
	  COUNT(CASE WHEN COMPLETESET = '2' THEN GRADE ELSE NULL END)/2 AS "GRADES_COMPLETE_2_4",
	  COUNT(CASE WHEN COMPLETESET = '3' THEN GRADE ELSE NULL END)/3 AS "GRADES_COMPLETE_1_2_3",
	  COUNT(CASE WHEN COMPLETESET = '4' THEN GRADE ELSE NULL END)/4 AS "GRADES_COMPLETE_1_2_3_4",
	  COUNT(CASE WHEN COMPLETESET = '5' THEN GRADE ELSE NULL END)/2 AS "GRADES_COMPLETE_2_3",
	  COUNT(CASE WHEN COMPLETESET = '6' THEN GRADE ELSE NULL END)/3 AS "GRADES_COMPLETE_2_3_4",
	  COUNT(CASE WHEN COMPLETESET = '7' THEN GRADE ELSE NULL END)/2 AS "GRADES_COMPLETE_1_2",
	  COUNT(CASE WHEN COMPLETESET = '8' THEN GRADE ELSE NULL END)/2 AS "GRADES_COMPLETE_1_4",
	  COUNT(CASE WHEN COMPLETESET = '9' THEN GRADE ELSE NULL END)/2 AS "GRADES_COMPLETE_3_4",
	  COUNT(CASE WHEN COMPLETESET = '10' THEN GRADE ELSE NULL END)/3 AS "GRADES_COMPLETE_1_2_4",
	  COUNT(CASE WHEN COMPLETESET = '11' THEN GRADE ELSE NULL END)/3 AS "GRADES_COMPLETE_1_3_4"	  
FROM (SELECT 
	 		GRADES.GRADEGROUPFK AS "GRADE",
	 		GRADES.LITER AS "LITER",
			GRADES.CLASSCOMPLETESET AS "COMPLETESET"
	 FROM SCHOOLS
	 INNER JOIN GRADES ON GRADES.SYS_GUIDFK = SCHOOLS.SYS_GUID
	 INNER JOIN PARTICIPANTS ON GRADES.SYS_GUID = PARTICIPANTS.SYS_GUIDFK
	 WHERE GRADES.CLASSCOMPLETE = '1'
	 GROUP BY GRADES.GRADEGROUPFK,GRADES.LITER,GRADES.CLASSCOMPLETESET) AS T1) AS T2
