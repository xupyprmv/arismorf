SELECT
	  COUNT(CASE WHEN COUNTER < 14 THEN GRADE ELSE NULL END) AS "GR_LESS14"
FROM (SELECT 
	 		GRADES.GRADEGROUPFK AS "GRADE",
	 		GRADES.LITER AS "LITER",  
	 		COUNT(PARTICIPANTS.SYS_GUID) AS "COUNTER"  
	 FROM SCHOOLS
	 INNER JOIN GRADES ON GRADES.SYS_GUIDFK = SCHOOLS.SYS_GUID
	 INNER JOIN PARTICIPANTS ON GRADES.SYS_GUID = PARTICIPANTS.SYS_GUIDFK
	 WHERE (GRADES.CLASSCOMPLETE = '0' or GRADES.CLASSCOMPLETE is null) AND SCHOOLS.TOWNTYPEFK IN ('1','2') AND PARTICIPANTS.PEOPLELEAVING IS NULL
	 GROUP BY GRADES.GRADEGROUPFK,GRADES.LITER) AS T
