SELECT 
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '1' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-2",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '2' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-4",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '3' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-5",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '4' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-6",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '5' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-7",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '6' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-8",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '7' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-9",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '8' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-10",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '9' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-11",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '10' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-12",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '11' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-13",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '12' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "5-14" 
FROM SCHOOLS
INNER JOIN GRADES ON SCHOOLS.SYS_GUID = GRADES.SYS_GUIDFK
INNER JOIN PARTICIPANTS ON PARTICIPANTS.SYS_GUIDFK = GRADES.SYS_GUID
LEFT JOIN ADDITIONALDATA ON ADDITIONALDATA.SYS_GUIDFK = SCHOOLS.SYS_GUID
WHERE GRADES.CLASSREMARK NOT IN ('4','5') AND PARTICIPANTS.PEOPLELEAVING IS NULL AND PARTICIPANTS.SEX = 'Æ'
