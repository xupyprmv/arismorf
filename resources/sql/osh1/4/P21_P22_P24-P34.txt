SELECT 
	   ADDITIONALDATA.PREPARCLASS_CNT AS "1-1",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '1' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-2",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '2' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-4",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '3' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-5",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '4' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-6",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '5' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-7",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '6' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-8",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '7' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-9",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '8' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-10",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '9' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-11",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '10' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-12",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '11' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-13",
	   COUNT(CASE WHEN GRADES.GRADEGROUPFK = '12' THEN GRADES.SYS_GUID ELSE NULL END) AS "1-14"	   
FROM SCHOOLS
INNER JOIN GRADES ON SCHOOLS.SYS_GUID = GRADES.SYS_GUIDFK
INNER JOIN ADDITIONALDATA ON ADDITIONALDATA.SYS_GUIDFK = SCHOOLS.SYS_GUID
WHERE GRADES.CLASSREMARK NOT IN ('4','5')
GROUP BY ADDITIONALDATA.PREPARCLASS_CNT
