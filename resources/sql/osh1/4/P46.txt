SELECT 
	   COUNT(PARTICIPANTS.SYS_GUID)
FROM SCHOOLS
INNER JOIN GRADES ON SCHOOLS.SYS_GUID = GRADES.SYS_GUIDFK
INNER JOIN PARTICIPANTS ON PARTICIPANTS.SYS_GUIDFK = GRADES.SYS_GUID
LEFT JOIN ADDITIONALDATA ON ADDITIONALDATA.SYS_GUIDFK = SCHOOLS.SYS_GUID
WHERE GRADES.CLASSREMARK IN ('4','5') AND PARTICIPANTS.PEOPLELEAVING IS NULL
