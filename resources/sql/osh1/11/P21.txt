SELECT
	  COUNT(CASE WHEN (ADDITIONALDATA.COUNCIL_BOARD = '1' OR ADDITIONALDATA.COUNCIL_OPERATING = '1' OR ADDITIONALDATA.COUNCIL_SCHOOL = '1' OR ADDITIONALDATA.COUNCIL_OTHER IS NOT NULL) THEN SCHOOLS.SYS_GUID ELSE NULL END) 
FROM ADDITIONALDATA
INNER JOIN SCHOOLS ON SCHOOLS.SYS_GUID = ADDITIONALDATA.SYS_GUIDFK
