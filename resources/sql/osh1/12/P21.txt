SELECT
	  CASE WHEN COUNTER = '0' THEN COUNTER ELSE COUNT(COUNTER) END
FROM (SELECT 
	   		 COUNT(CASE WHEN PARTICIPANTS.PAID_EDUC_SERVICES = '1' THEN PARTICIPANTS.SYS_GUID ELSE NULL END) AS "COUNTER"
	   FROM GRADES
	   INNER JOIN PARTICIPANTS ON PARTICIPANTS.SYS_GUIDFK = GRADES.SYS_GUID
	   INNER JOIN SCHOOLS ON SCHOOLS.SYS_GUID = GRADES.SYS_GUIDFK) AS T
GROUP BY COUNTER
