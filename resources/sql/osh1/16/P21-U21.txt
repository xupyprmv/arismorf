SELECT
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASSINITIAL_FOREIGNLANG = '1' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASSINITIAL_FOREIGNLANG = '1' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASSINITIAL_FOREIGNLANG = '2' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASSINITIAL_FOREIGNLANG = '2' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(DISTINCT CASE WHEN EGEEXTPARTICIPANTS.CLASSINITIAL_FOREIGNLANG = '3' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN EGEEXTPARTICIPANTS.CLASSINITIAL_FOREIGNLANG = '3' THEN PARTICIPANTS.SYS_GUID ELSE NULL END)	  	  	 
FROM PARTICIPANTS
INNER JOIN EGEEXTPARTICIPANTS ON EGEEXTPARTICIPANTS.SYS_GUIDFK = PARTICIPANTS.SYS_GUID
INNER JOIN GRADES ON GRADES.SYS_GUID = PARTICIPANTS.SYS_GUIDFK
