SELECT
	  MATBASE.BUILD_CNT,
	  MATBASE.BUILD_AREA,
	  MATBASE.CLASS_CNT,
	  MATBASE.CLASS_AREA,
	  MATBASE.CLS_MASTER_CNT,
	  MATBASE.TRACTOR_CNT,
	  MATBASE.CLS_PHYSCULT,
	  MATBASE.CLS_SWIMPOOL,
	  MATBASE.CLS_ASSEMBLY,
	  MATBASE.CLS_MUSEUM,
	  MATBASE.AGRICULTURE_AREA,
	  MATBASE.LIB_STORECNT,
	  MATBASE.LIB_SCHBOOK
FROM MATBASE
INNER JOIN SCHOOLS ON SCHOOLS.SYS_GUID = MATBASE.SYS_GUIDFK
