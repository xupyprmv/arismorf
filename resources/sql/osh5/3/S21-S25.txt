SELECT
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK IN ('1','2','3','4') THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK IN ('5','6','7','8') THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK = '9' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK = '10' THEN PARTICIPANTS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK IN ('11','12') THEN PARTICIPANTS.SYS_GUID ELSE NULL END)	   
FROM GRADES
INNER JOIN PARTICIPANTS ON PARTICIPANTS.SYS_GUIDFK = GRADES.SYS_GUID
