SELECT
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK IN ('1','2','3','4') THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK IN ('5','6','7','8') THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK = '9' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK = '10' THEN GRADES.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN GRADES.GRADEGROUPFK IN ('11','12') THEN GRADES.SYS_GUID ELSE NULL END)	   
FROM GRADES
WHERE GRADES.LITER = 'гн'
