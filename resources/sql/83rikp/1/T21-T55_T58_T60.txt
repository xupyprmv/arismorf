	  COUNT(CASE WHEN TEACHERS.WORK_PEDAGOGWORKER IN ('99') THEN TEACHERS.SYS_GUID ELSE NULL END),
	  
	  COUNT(CASE WHEN TEACHERS.WORK_PERS_UCHEB = '1' THEN TEACHERS.SYS_GUID ELSE NULL END),
	  COUNT(CASE WHEN TEACHERS.WORK_PERS_SERV = '1' THEN TEACHERS.SYS_GUID ELSE NULL END)	  	  	   
FROM TEACHERS
